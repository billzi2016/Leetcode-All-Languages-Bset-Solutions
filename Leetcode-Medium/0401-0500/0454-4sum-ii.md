```cpp
class Solution {
public:
    int fourSumCount(vector<int>& nums1, vector<int>& nums2, vector<int>& nums3, vector<int>& nums4) {
        unordered_map<int, int> sumMap;
        for (int a : nums1) {
            for (int b : nums2) {
                ++sumMap[a + b];
            }
        }
        int result = 0;
        for (int c : nums3) {
            for (int d : nums4) {
                auto it = sumMap.find(-(c + d));
                if (it != sumMap.end()) {
                    result += it->second;
                }
            }
        }
        return result;
    }
};
```

```c
#include <stdlib.h>

static int cmp_int(const void *a, const void *b) {
    int va = *(const int *)a;
    int vb = *(const int *)b;
    if (va < vb) return -1;
    if (va > vb) return 1;
    return 0;
}

int fourSumCount(int* nums1, int nums1Size,
                 int* nums2, int nums2Size,
                 int* nums3, int nums3Size,
                 int* nums4, int nums4Size) {
    int n = nums1Size; // all sizes are equal
    int totalPairs = n * n;
    
    int *AB = (int *)malloc(sizeof(int) * totalPairs);
    int *CD = (int *)malloc(sizeof(int) * totalPairs);
    if (!AB || !CD) {
        free(AB);
        free(CD);
        return 0;
    }
    
    int idx = 0;
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            AB[idx++] = nums1[i] + nums2[j];
    
    idx = 0;
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            CD[idx++] = nums3[i] + nums4[j];
    
    qsort(AB, totalPairs, sizeof(int), cmp_int);
    qsort(CD, totalPairs, sizeof(int), cmp_int);
    
    long long result = 0;
    int i = 0;
    int j = totalPairs - 1;
    while (i < totalPairs && j >= 0) {
        long long sum = (long long)AB[i] + CD[j];
        if (sum == 0) {
            int aVal = AB[i];
            int cntA = 0;
            while (i < totalPairs && AB[i] == aVal) {
                ++cntA;
                ++i;
            }
            int cVal = CD[j];
            int cntC = 0;
            while (j >= 0 && CD[j] == cVal) {
                ++cntC;
                --j;
            }
            result += (long long)cntA * cntC;
        } else if (sum < 0) {
            ++i;
        } else { // sum > 0
            --j;
        }
    }
    
    free(AB);
    free(CD);
    return (int)result;
}
```

```csharp
public class Solution {
    public int FourSumCount(int[] nums1, int[] nums2, int[] nums3, int[] nums4) {
        var sumFreq = new Dictionary<int, int>();
        foreach (int a in nums1) {
            foreach (int b in nums2) {
                int s = a + b;
                if (sumFreq.ContainsKey(s))
                    sumFreq[s]++;
                else
                    sumFreq[s] = 1;
            }
        }

        long count = 0;
        foreach (int c in nums3) {
            foreach (int d in nums4) {
                int target = -(c + d);
                if (sumFreq.TryGetValue(target, out int freq))
                    count += freq;
            }
        }

        return (int)count;
    }
}
```

```racket
(define/contract (four-sum-count nums1 nums2 nums3 nums4)
  (-> (listof exact-integer?) (listof exact-integer?) (listof exact-integer?) (listof exact-integer?) exact-integer?)
  (let ((hash (make-hash)))
    ;; sums of nums1 and nums2
    (for* ([a nums1] [b nums2])
      (let ((s (+ a b)))
        (hash-set! hash s (+ 1 (hash-ref hash s 0)))))
    (define count 0)
    ;; complement sums from nums3 and nums4
    (for* ([c nums3] [d nums4])
      (let ((need (- (+ c d))))
        (when (hash-has-key? hash need)
          (set! count (+ count (hash-ref hash need))))))
    count))
```

```elixir
defmodule Solution do
  @spec four_sum_count(nums1 :: [integer], nums2 :: [integer], nums3 :: [integer], nums4 :: [integer]) :: integer
  def four_sum_count(nums1, nums2, nums3, nums4) do
    sum_counts =
      Enum.reduce(nums1, %{}, fn a, acc ->
        Enum.reduce(nums2, acc, fn b, inner_acc ->
          s = a + b
          Map.update(inner_acc, s, 1, &(&1 + 1))
        end)
      end)

    Enum.reduce(nums3, 0, fn c, total ->
      Enum.reduce(nums4, total, fn d, acc_total ->
        target = -(c + d)

        case Map.get(sum_counts, target) do
          nil -> acc_total
          cnt -> acc_total + cnt
        end
      end)
    end)
  end
end
```
