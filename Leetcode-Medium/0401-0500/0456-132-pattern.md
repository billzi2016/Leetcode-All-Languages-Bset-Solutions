# 0456. 132 Pattern

## Cpp

```cpp
class Solution {
public:
    bool find132pattern(vector<int>& nums) {
        int n = nums.size();
        if (n < 3) return false;
        vector<int> mins(n);
        mins[0] = nums[0];
        for (int i = 1; i < n; ++i)
            mins[i] = min(mins[i - 1], nums[i]);
        
        vector<int> st; // monotonic decreasing stack
        for (int j = n - 1; j >= 0; --j) {
            if (nums[j] > mins[j]) {
                while (!st.empty() && st.back() <= mins[j])
                    st.pop_back();
                if (!st.empty() && st.back() < nums[j])
                    return true;
                st.push_back(nums[j]);
            }
        }
        return false;
    }
};
```

```javascript
    const minArr = new Array(n);
```

```dart
class Solution {
  bool find132pattern(List<int> nums) {
    int n = nums.length;
    if (n < 3) return false;

    // minLeft[i] stores the minimum value from nums[0] to nums[i]
    List<int> minLeft = List.filled(n, 0);
    int curMin = nums[0];
    for (int i = 0; i < n; i++) {
      if (nums[i] < curMin) curMin = nums[i];
      minLeft[i] = curMin;
    }

    // Stack to keep potential "2" values in decreasing order
    List<int> stack = [];

    for (int j = n - 1; j >= 0; j--) {
      if (nums[j] > minLeft[j]) {
        while (stack.isNotEmpty && stack.last <= minLeft[j]) {
          stack.removeLast();
        }
        if (stack.isNotEmpty && stack.last < nums[j]) {
          return true;
        }
        stack.add(nums[j]);
      }
    }

    return false;
  }
}
```

```go
				stack = stack[:len(stack)-1]
			}
			if len(stack) > 0 && stack[len(stack)-1] < nums[j] {
				return true
			}
			stack = append(stack, nums[j])
		}
	}
	return false
```

```ruby
    end
  end
```

```rust
impl Solution {
    pub fn find132pattern(nums: Vec<i32>) -> bool {
        let n = nums.len();
        if n < 3 {
            return false;
        }
        // mins[i] is the minimum value in nums[0..=i]
        let mut mins = vec![0i32; n];
        mins[0] = nums[0];
        for i in 1..n {
            mins[i] = std::cmp::min(mins[i - 1], nums[i]);
        }

        // stack will store candidate "k" values in decreasing order
        let mut stack: Vec<i32> = Vec::new();

        for j in (0..n).rev() {
            if nums[j] > mins[j] {
                while let Some(&top) = stack.last() {
                    if top <= mins[j] {
                        stack.pop();
                    } else {
                        break;
                    }
                }
                if let Some(&top) = stack.last() {
                    if top < nums[j] {
                        return true;
                    }
                }
                stack.push(nums[j]);
            }
        }

        false
    }
}
```

```erlang
-spec find132pattern([integer()]) -> boolean().
find132pattern(Nums) ->
    PrefixMins = prefix_mins(Nums),
    RevNums = lists:reverse(Nums),
    RevMins = lists:reverse(PrefixMins),
    iter_rev(RevNums, RevMins, []).

prefix_mins([]) -> [];
prefix_mins([H|T]) ->
    prefix_mins(T, H, [H]).

prefix_mins([], _CurMin, Acc) ->
    lists:reverse(Acc);
prefix_mins([X|Xs], CurMin, Acc) ->
    NewMin = if X < CurMin -> X; true -> CurMin end,
    prefix_mins(Xs, NewMin, [NewMin | Acc]).

iter_rev([], [], _Stack) -> false;
iter_rev([N|Ns], [M|Ms], Stack) ->
    case N > M of
        true ->
            NewStack = pop_while(Stack, fun (X) -> X =< M end),
            case NewStack of
                [Top | _] when Top < N -> true;
                _ -> iter_rev(Ns, Ms, [N | NewStack])
            end;
        false ->
            iter_rev(Ns, Ms, Stack)
    end.

pop_while([], _Pred) -> [];
pop_while([H|T], Pred) ->
    case Pred(H) of
        true -> pop_while(T, Pred);
        false -> [H | T]
    end.
```

```elixir
defmodule Solution do
  @spec find132pattern(nums :: [integer]) :: boolean
  def find132pattern(nums) do
    third = -1_000_000_001

    {found, _stack, _third} =
      Enum.reduce(Enum.reverse(nums), {false, [], third}, fn x, {found, stack, third} ->
        if found do
          {true, stack, third}
        else
          cond do
            x < third ->
              {true, stack, third}

            true ->
              {new_stack, new_third} = pop_while_greater(x, stack, third)
              {false, [x | new_stack], new_third}
          end
        end
      end)

    found
  end

  defp pop_while_greater(_val, [], third), do: {[], third}

  defp pop_while_greater(val, [h | t] = _stack, _third) when val > h do
    # pop and update third to the popped value
    pop_while_greater(val, t, h)
  end
```
