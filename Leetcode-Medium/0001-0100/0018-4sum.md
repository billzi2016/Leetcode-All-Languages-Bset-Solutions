```cpp
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    vector<vector<int>> fourSum(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        return kSum(nums, 0, 4, (long long)target);
    }
private:
    vector<vector<int>> kSum(const vector<int>& nums, int start, int k, long long target) {
        vector<vector<int>> res;
        int n = nums.size();
        if (k == 2) return twoSum(nums, start, target);
        for (int i = start; i <= n - k; ++i) {
            if (i > start && nums[i] == nums[i-1]) continue; // skip duplicates
            // early termination checks
            long long minPossible = (long long)nums[i];
            for (int j = 1; j < k; ++j) minPossible += (long long)nums[i + j];
            if (minPossible > target) break;
            long long maxPossible = (long long)nums[i];
            for (int j = 1; j < k; ++j) maxPossible += (long long)nums[n - 1 - j + 1];
            if (maxPossible < target) continue;

            auto subsets = kSum(nums, i + 1, k - 1, target - nums[i]);
            for (auto& subset : subsets) {
                vector<int> comb;
                comb.reserve(k);
                comb.push_back(nums[i]);
                comb.insert(comb.end(), subset.begin(), subset.end());
                res.push_back(move(comb));
            }
        }
        return res;
    }

    vector<vector<int>> twoSum(const vector<int>& nums, int start, long long target) {
        vector<vector<int>> res;
        int lo = start, hi = (int)nums.size() - 1;
        while (lo < hi) {
            long long sum = (long long)nums[lo] + nums[hi];
            if (sum < target) {
                ++lo;
            } else if (sum > target) {
                --hi;
            } else {
                res.push_back({nums[lo], nums[hi]});
                int curLo = nums[lo], curHi = nums[hi];
                while (lo < hi && nums[lo] == curLo) ++lo;
                while (lo < hi && nums[hi] == curHi) --hi;
            }
        }
        return res;
    }
};
```

```java
class Solution {
    public List<List<Integer>> fourSum(int[] nums, int target) {
        Arrays.sort(nums);
        return kSum(nums, 0, 4, (long) target);
    }

    private List<List<Integer>> kSum(int[] nums, int start, int k, long target) {
        List<List<Integer>> res = new ArrayList<>();
        int n = nums.length;
        if (k == 2) {
            return twoSum(nums, start, target);
        }
        // Prune: not enough numbers
        if (start >= n) return res;
        // Minimum possible sum with k smallest numbers
        long minSum = 0;
        for (int i = 0; i < k; i++) {
            minSum += nums[start + i];
        }
        // Maximum possible sum with k largest numbers
        long maxSum = 0;
        for (int i = 0; i < k; i++) {
            maxSum += nums[n - 1 - i];
        }
        if (target < minSum || target > maxSum) return res;

        for (int i = start; i <= n - k; i++) {
            if (i > start && nums[i] == nums[i - 1]) continue;
            long newTarget = target - nums[i];
            List<List<Integer>> sub = kSum(nums, i + 1, k - 1, newTarget);
            for (List<Integer> list : sub) {
                List<Integer> cur = new ArrayList<>(k);
                cur.add(nums[i]);
                cur.addAll(list);
                res.add(cur);
            }
        }
        return res;
    }

    private List<List<Integer>> twoSum(int[] nums, int start, long target) {
        List<List<Integer>> res = new ArrayList<>();
        int lo = start, hi = nums.length - 1;
        while (lo < hi) {
            long sum = (long) nums[lo] + nums[hi];
            if (sum < target) {
                lo++;
            } else if (sum > target) {
                hi--;
            } else {
                List<Integer> pair = new ArrayList<>(2);
                pair.add(nums[lo]);
                pair.add(nums[hi]);
                res.add(pair);
                lo++;
                hi--;
                while (lo < hi && nums[lo] == nums[lo - 1]) lo++;
                while (lo < hi && nums[hi] == nums[hi + 1]) hi--;
            }
        }
        return res;
    }
}
```

```python
class Solution(object):
    def fourSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[List[int]]
        """
        nums.sort()
        n = len(nums)

        def kSum(start, k, tgt):
            res = []
            # early termination
            if start >= n or k * nums[start] > tgt or k * nums[-1] < tgt:
                return res
            if k == 2:
                lo, hi = start, n - 1
                while lo < hi:
                    s = nums[lo] + nums[hi]
                    if s == tgt:
                        res.append([nums[lo], nums[hi]])
                        lo += 1
                        hi -= 1
                        while lo < hi and nums[lo] == nums[lo - 1]:
                            lo += 1
                        while lo < hi and nums[hi] == nums[hi + 1]:
                            hi -= 1
                    elif s < tgt:
                        lo += 1
                    else:
                        hi -= 1
                return res

            for i in range(start, n - k + 1):
                if i > start and nums[i] == nums[i - 1]:
                    continue
                # prune with current number
                if nums[i] * k > tgt:
                    break
                if nums[i] + (k - 1) * nums[-1] < tgt:
                    continue
                for subset in kSum(i + 1, k - 1, tgt - nums[i]):
                    res.append([nums[i]] + subset)
            return res

        return kSum(0, 4, target)
```

```c
#include <stdlib.h>

static int cmp_int(const void *a, const void *b) {
    int av = *(const int *)a;
    int bv = *(const int *)b;
    if (av < bv) return -1;
    if (av > bv) return 1;
    return 0;
}

/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
int** fourSum(int* nums, int numsSize, int target, int* returnSize, int*** returnColumnSizes) {
    (void)returnColumnSizes; // silence unused warning if not needed
    *returnSize = 0;
    if (numsSize < 4) return NULL;

    qsort(nums, numsSize, sizeof(int), cmp_int);

    int capacity = 128;
    int **result = (int **)malloc(capacity * sizeof(int *));
    if (!result) return NULL;

    for (int i = 0; i < numsSize - 3; ++i) {
        if (i > 0 && nums[i] == nums[i - 1]) continue;
        // early termination checks
        long long min1 = (long long)nums[i] + nums[i + 1] + nums[i + 2] + nums[i + 3];
        if (min1 > target) break;
        long long max1 = (long long)nums[i] + nums[numsSize - 1] + nums[numsSize - 2] + nums[numsSize - 3];
        if (max1 < target) continue;

        for (int j = i + 1; j < numsSize - 2; ++j) {
            if (j > i + 1 && nums[j] == nums[j - 1]) continue;
            long long min2 = (long long)nums[i] + nums[j] + nums[j + 1] + nums[j + 2];
            if (min2 > target) break;
            long long max2 = (long long)nums[i] + nums[j] + nums[numsSize - 1] + nums[numsSize - 2];
            if (max2 < target) continue;

            int left = j + 1;
            int right = numsSize - 1;
            while (left < right) {
                long long sum = (long long)nums[i] + nums[j] + nums[left] + nums[right];
                if (sum == target) {
                    if (*returnSize >= capacity) {
                        capacity <<= 1;
                        int **tmp = (int **)realloc(result, capacity * sizeof(int *));
                        if (!tmp) {
                            // free already allocated memory before returning
                            for (int k = 0; k < *returnSize; ++k) free(result[k]);
                            free(result);
                            return NULL;
                        }
                        result = tmp;
                    }
                    int *quad = (int *)malloc(4 * sizeof(int));
                    if (!quad) {
                        // free already allocated memory before returning
                        for (int k = 0; k < *returnSize; ++k) free(result[k]);
                        free(result);
                        return NULL;
                    }
                    quad[0] = nums[i];
                    quad[1] = nums[j];
                    quad[2] = nums[left];
                    quad[3] = nums[right];
                    result[*returnSize] = quad;
                    (*returnSize)++;

                    // skip duplicates
                    int curLeft = nums[left];
                    int curRight = nums[right];
                    while (left < right && nums[left] == curLeft) ++left;
                    while (left < right && nums[right] == curRight) --right;
                } else if (sum < target) {
                    ++left;
                } else {
                    --right;
                }
            }
        }
    }

    if (*returnSize == 0) {
        free(result);
        return NULL;
    }

    // allocate column sizes
    int *colSizes = (int *)malloc((*returnSize) * sizeof(int));
    for (int i = 0; i < *returnSize; ++i) colSizes[i] = 4;
    *returnColumnSizes = &colSizes;

    return result;
}
```

```csharp
public class Solution {
    public IList<IList<int>> FourSum(int[] nums, int target) {
        Array.Sort(nums);
        return KSum(nums, 0, 4, (long)target);
    }

    private IList<IList<int>> KSum(int[] nums, int start, int k, long target) {
        var res = new List<IList<int>>();
        if (k == 2) {
            return TwoSum(nums, start, target);
        }
        // Prune: not enough numbers
        if (start >= nums.Length) return res;
        // Minimum possible sum with the smallest k numbers
        long minSum = 0;
        for (int i = 0; i < k; i++) {
            if (start + i >= nums.Length) return res;
            minSum += nums[start + i];
        }
        if (minSum > target) return res;
        // Maximum possible sum with the largest k numbers
        long maxSum = 0;
        for (int i = 0; i < k; i++) {
            maxSum += nums[nums.Length - 1 - i];
        }
        if (maxSum < target) return res;

        for (int i = start; i <= nums.Length - k; i++) {
            if (i > start && nums[i] == nums[i - 1]) continue;
            var subsets = KSum(nums, i + 1, k - 1, target - nums[i]);
            foreach (var subset in subsets) {
                var list = new List<int>();
                list.Add(nums[i]);
                list.AddRange(subset);
                res.Add(list);
            }
        }
        return res;
    }

    private IList<IList<int>> TwoSum(int[] nums, int start, long target) {
        var res = new List<IList<int>>();
        int lo = start;
        int hi = nums.Length - 1;
        while (lo < hi) {
            long sum = (long)nums[lo] + nums[hi];
            if (sum == target) {
                var pair = new List<int> { nums[lo], nums[hi] };
                res.Add(pair);
                lo++;
                hi--;
                while (lo < hi && nums[lo] == nums[lo - 1]) lo++;
                while (lo < hi && nums[hi] == nums[hi + 1]) hi--;
            } else if (sum < target) {
                lo++;
            } else {
                hi--;
            }
        }
        return res;
    }
}
```

```swift
class Solution {
    func fourSum(_ nums: [Int], _ target: Int) -> [[Int]] {
        let sorted = nums.sorted()
        let n = sorted.count
        if n < 4 { return [] }
        
        func kSum(_ start: Int, _ k: Int, _ target: Int64) -> [[Int]] {
            var res = [[Int]]()
            if start >= n { return res }
            
            // Early termination checks
            let minPossible = Int64(sorted[start]) * Int64(k)
            let maxPossible = Int64(sorted[n - 1]) * Int64(k)
            if target < minPossible || target > maxPossible {
                return res
            }
            
            if k == 2 {
                var lo = start
                var hi = n - 1
                while lo < hi {
                    let sum = Int64(sorted[lo]) + Int64(sorted[hi])
                    if sum == target {
                        res.append([sorted[lo], sorted[hi]])
                        let leftVal = sorted[lo]
                        repeat { lo += 1 } while lo < hi && sorted[lo] == leftVal
                        let rightVal = sorted[hi]
                        repeat { hi -= 1 } while lo < hi && sorted[hi] == rightVal
                    } else if sum < target {
                        lo += 1
                    } else {
                        hi -= 1
                    }
                }
                return res
            }
            
            for i in start..<(n - k + 1) {
                if i > start && sorted[i] == sorted[i - 1] { continue }
                let current = Int64(sorted[i])
                let subResults = kSum(i + 1, k - 1, target - current)
                for var subset in subResults {
                    subset.insert(sorted[i], at: 0)
                    res.append(subset)
                }
            }
            return res
        }
        
        return kSum(0, 4, Int64(target))
    }
```

```kotlin
class Solution {
    fun fourSum(nums: IntArray, target: Int): List<List<Int>> {
        val sorted = nums.sorted()
        val n = sorted.size
        val arr = IntArray(n) { sorted[it] }

        fun twoSum(start: Int, targetLong: Long): MutableList<List<Int>> {
            var lo = start
            var hi = n - 1
            val res = mutableListOf<List<Int>>()
            while (lo < hi) {
                val sum = arr[lo].toLong() + arr[hi]
                when {
                    sum < targetLong -> lo++
                    sum > targetLong -> hi--
                    else -> {
                        res.add(listOf(arr[lo], arr[hi]))
                        val leftVal = arr[lo]
                        while (lo < hi && arr[lo] == leftVal) lo++
                        val rightVal = arr[hi]
                        while (lo < hi && arr[hi] == rightVal) hi--
                    }
                }
            }
            return res
        }

        fun kSum(start: Int, k: Int, targetLong: Long): MutableList<List<Int>> {
            val res = mutableListOf<List<Int>>()
            if (start >= n) return res

            // early termination checks
            val minPossible = arr[start].toLong() * k
            if (minPossible > targetLong) return res
            val maxPossible = arr[n - 1].toLong() * k
            if (maxPossible < targetLong) return res

            if (k == 2) {
                return twoSum(start, targetLong)
            }

            var i = start
            while (i <= n - k) {
                if (i > start && arr[i] == arr[i - 1]) {
                    i++
                    continue
                }
                val subLists = kSum(i + 1, k - 1, targetLong - arr[i])
                for (sub in subLists) {
                    val list = mutableListOf<Int>()
                    list.add(arr[i])
                    list.addAll(sub)
                    res.add(list)
                }
                i++
            }
            return res
        }

        return kSum(0, 4, target.toLong())
    }
}
```

```go
	minSum, maxSum := int64(0), int64(0)
	for i := 0; i < k; i++ {
		minSum += int64(nums[start+i])
		maxSum += int64(nums[n-1-i])
	}
	if minSum > target || maxSum < target {
		return res
	}
	for i := start; i <= n-k; i++ {
		if i > start && nums[i] == nums[i-1] {
			continue
		}
		subsets := kSum(nums, i+1, k-1, target-int64(nums[i]))
		for _, subset := range subsets {
			comb := make([]int, 0, k)
			comb = append(comb, nums[i])
			comb = append(comb, subset...)
			res = append(res, comb)
		}
	}
	return res
}

func twoSum(nums []int, start int, target int64) [][]int {
	lo, hi := start, len(nums)-1
	var res [][]int
	for lo < hi {
		sum := int64(nums[lo]) + int64(nums[hi])
		if sum < target {
			lo++
		} else if sum > target {
			hi--
		} else {
			res = append(res, []int{nums[lo], nums[hi]})
			curLo, curHi := nums[lo], nums[hi]
			for lo < hi && nums[lo] == curLo {
				lo++
			}
			for lo < hi && nums[hi] == curHi {
				hi--
			}
		}
	}
	return res
}
```

```ruby
def four_sum(nums, target)
  nums.sort!
  n = nums.length
  result = []

  (0...n - 3).each do |i|
    next if i > 0 && nums[i] == nums[i - 1]

    (i + 1...n - 2).each do |j|
      next if j > i + 1 && nums[j] == nums[j - 1]

      left = j + 1
      right = n - 1

      while left < right
        sum = nums[i] + nums[j] + nums[left] + nums[right]
        if sum == target
          result << [nums[i], nums[j], nums[left], nums[right]]
          left += 1
          right -= 1
          while left < right && nums[left] == nums[left - 1]
            left += 1
          end
          while left < right && nums[right] == nums[right + 1]
            right -= 1
          end
        elsif sum < target
          left += 1
        else
          right -= 1
        end
      end
    end
  end

  result
```

```scala
object Solution {
    def fourSum(nums: Array[Int], target: Int): List[List[Int]] = {
        val n = nums.length
        if (n < 4) return Nil
        val sorted = nums.sorted

        def twoSum(start: Int, targetLong: Long): List[List[Int]] = {
            var lo = start
            var hi = n - 1
            var res = List.empty[List[Int]]
            while (lo < hi) {
                val sum = sorted(lo).toLong + sorted(hi).toLong
                if (sum < targetLong) {
                    lo += 1
                } else if (sum > targetLong) {
                    hi -= 1
                } else {
                    res = List(sorted(lo), sorted(hi)) :: res
                    val curLo = sorted(lo)
                    val curHi = sorted(hi)
                    while (lo < hi && sorted(lo) == curLo) lo += 1
                    while (lo < hi && sorted(hi) == curHi) hi -= 1
                }
            }
            res.reverse
        }

        def kSum(start: Int, k: Int, targetLong: Long): List[List[Int]] = {
            if (k == 2) return twoSum(start, targetLong)
            var res = List.empty[List[Int]]
            for (i <- start until n - k + 1) {
                if (i > start && sorted(i) == sorted(i - 1)) {
                    // skip duplicate
                } else {
                    val current = sorted(i)
                    val subLists = kSum(i + 1, k - 1, targetLong - current.toLong)
                    for (sub <- subLists) {
                        res = (current :: sub) :: res
                    }
                }
            }
            res.reverse
        }

        kSum(0, 4, target.toLong)
    }
}
```

```rust
impl Solution {
    pub fn four_sum(nums: Vec<i32>, target: i32) -> Vec<Vec<i32>> {
        let mut a: Vec<i64> = nums.iter().map(|&x| x as i64).collect();
        a.sort_unstable();

        fn k_sum(start: usize, k: usize, target: i64, nums: &Vec<i64>) -> Vec<Vec<i64>> {
            let n = nums.len();
            if start + k > n {
                return Vec::new();
            }
            // pruning with min/max possible sums
            let min_sum: i64 = (0..k).map(|i| nums[start + i]).sum();
            if min_sum > target {
                return Vec::new();
            }
            let max_sum: i64 = (0..k).map(|i| nums[n - 1 - i]).sum();
            if max_sum < target {
                return Vec::new();
            }

            let mut res = Vec::new();

            if k == 2 {
                let mut lo = start;
                let mut hi = n - 1;
                while lo < hi {
                    let sum = nums[lo] + nums[hi];
                    if sum == target {
                        res.push(vec![nums[lo], nums[hi]]);
                        let cur_lo = nums[lo];
                        while lo < hi && nums[lo] == cur_lo {
                            lo += 1;
                        }
                        let cur_hi = nums[hi];
                        while lo < hi && nums[hi] == cur_hi {
                            hi -= 1;
                        }
                    } else if sum < target {
                        lo += 1;
                    } else {
                        hi -= 1;
                    }
                }
            } else {
                for i in start..=n - k {
                    if i > start && nums[i] == nums[i - 1] {
                        continue;
                    }
                    let cur = nums[i];
                    let sub = k_sum(i + 1, k - 1, target - cur, nums);
                    for mut subset in sub {
                        let mut vec = Vec::with_capacity(k);
                        vec.push(cur);
                        vec.append(&mut subset);
                        res.push(vec);
                    }
                }
            }

            res
        }

        let raw = k_sum(0, 4, target as i64, &a);
        raw.into_iter()
            .map(|v| v.into_iter().map(|x| x as i32).collect())
            .collect()
    }
}
```

## Racket

```racket
(define/contract (four-sum nums target)
  (-> (listof exact-integer?) exact-integer? (listof (listof exact-integer?)))
  (let* ((sorted (sort nums <))
         (v      (list->vector sorted))
         (n      (vector-length v)))

    ;; sum of cnt elements starting at idx
    (define (sum-from idx cnt)
      (let loop ((i idx) (c cnt) (s 0))
        (if (= c 0)
            s
            (loop (+ i 1) (- c 1) (+ s (vector-ref v i))))))

    ;; sum of cnt largest elements ending at endIdx (inclusive)
    (define (sum-to endIdx cnt)
      (let loop ((i endIdx) (c cnt) (s 0))
        (if (= c 0)
            s
            (loop (- i 1) (- c 1) (+ s (vector-ref v i))))))

    ;; two‑pointer for k = 2
    (define (two-sum start tgt)
      (let loop ((lo start) (hi (- n 1)) (acc '()))
        (if (>= lo hi)
            (reverse acc)
            (let* ((a (vector-ref v lo))
                   (b (vector-ref v hi))
                   (s (+ a b)))
              (cond
                [(< s tgt) (loop (+ lo 1) hi acc)]
                [(> s tgt) (loop lo (- hi 1) acc)]
                [else
                 (let ((pair (list a b)))
                   ;; skip duplicates on both sides
                   (let loop-lo ((l (+ lo 1)))
                     (if (and (< l hi) (= (vector-ref v l) a))
                         (loop-lo (+ l 1))
                         (let loop-hi ((h (- hi 1)))
                           (if (and (> h l) (= (vector-ref v h) b))
                               (loop-hi (- h 1))
                               (loop l h (cons pair acc)))))))]))))

    ;; recursive k‑sum
    (define (k-sum start k tgt)
      (cond
        [(or (> start n) (< (- n start) k)) '()] ; not enough numbers
        [else
         (let* ((min-s (sum-from start k))
                (max-s (sum-to (- n 1) k)))
           (if (or (< tgt min-s) (> tgt max-s))
               '()
               (if (= k 2)
                   (two-sum start tgt)
                   (let loop ((i start) (res '()))
                     (if (> i (- n k)) ; need at least k numbers left
                         (reverse res)
                         (let ((cur (vector-ref v i)))
                           (cond
                             [(and (> i start) (= cur (vector-ref v (- i 1)))) ; skip dup
                              (loop (+ i 1) res)]
                             [else
                              (let ((sub (k-sum (+ i 1) (- k 1) (- tgt cur))))
                                (define new-res
                                  (foldl (lambda (lst acc)
                                           (cons (cons cur lst) acc))
                                         res sub))
                                (loop (+ i 1) new-res))]))))))))))

    (k-sum 0 4 target)))
```

```erlang
k_sum([A|Rest], K, Target) when K > 2 ->
    Subsets = k_sum(Rest, K - 1, Target - A),
    WithA = [ [A | Sub] || Sub <- Subsets ],
    NextList = skip_equal(A, Rest),
    WithA ++ k_sum(NextList, K, Target).

pair_sum([A|Rest], Target) ->
    PairsWithA = pair_with_a(A, Rest, Target),
    NextList = skip_equal(A, Rest),
    PairsWithA ++ pair_sum(NextList, Target).

pair_with_a(_A, [], _Target) -> [];
pair_with_a(A, [B|Rest], Target) ->
    case A + B of
        Sum when Sum == Target ->
            Next = skip_equal(B, Rest),
            [[A, B] | pair_with_a(A, Next, Target)];
        _ ->
            pair_with_a(A, Rest, Target)
    end.

%% helper to skip consecutive duplicate values in a sorted list
skip_equal(_Val, []) -> [];
skip_equal(Val, [H|T]) when H == Val -> skip_equal(Val, T);
skip_equal(_Val, List) -> List.
```

## Elixir

```elixir
defmodule Solution do
  @spec four_sum(nums :: [integer], target :: integer) :: [[integer]]
  def four_sum(nums, target) do
    sorted = Enum.sort(nums)
    k_sum(sorted, 0, 4, target)
  end

  # General k-sum recursion
  defp k_sum(nums, start, k, target) do
    n = length(nums)

    cond do
      start >= n ->
        []

      k * Enum.at(nums, start) > target ->
        []

      k * Enum.at(nums, n - 1) < target ->
        []

      k == 2 ->
        two_sum(nums, start, target)

      true ->
        result =
          for i <- start..(n - k), reduce: [] do
            acc ->
              if i > start and Enum.at(nums, i) == Enum.at(nums, i - 1) do
                acc
              else
                current = Enum.at(nums, i)
                sub = k_sum(nums, i + 1, k - 1, target - current)

                new_lists =
                  for s <- sub do
                    [current | s]
                  end

                acc ++ new_lists
              end
          end

        result
    end
  end

  # Two-pointer two-sum on a sorted array
  defp two_sum(nums, start, target) do
    n = length(nums)
    two_sum_loop(nums, start, n - 1, target, [])
  end

  defp two_sum_loop(_nums, lo, hi, _target, acc) when lo >= hi do
    Enum.reverse(acc)
  end

  defp two_sum_loop(nums, lo, hi, target, acc) do
    left = Enum.at(nums, lo)
    right = Enum.at(nums, hi)
    sum = left + right

    cond do
      sum < target ->
        new_lo = lo + 1
        new_lo = skip_forward(nums, new_lo, hi, left)
        two_sum_loop(nums, new_lo, hi, target, acc)

      sum > target ->
        new_hi = hi - 1
        new_hi = skip_backward(nums, lo, new_hi, right)
        two_sum_loop(nums, lo, new_hi, target, acc)

      true ->
        pair = [left, right]
        new_lo = lo + 1
        new_lo = skip_forward(nums, new_lo, hi, left)
        new_hi = hi - 1
        new_hi = skip_backward(nums, lo, new_hi, right)
        two_sum_loop(nums, new_lo, new_hi, target, [pair | acc])
    end
  end

  defp skip_forward(_nums, idx, limit, _prev) when idx > limit, do: idx

  defp skip_forward(nums, idx, limit, prev) do
    if Enum.at(nums, idx) == prev do
      skip_forward(nums, idx + 1, limit, prev)
    else
      idx
    end
  end

  defp skip_backward(_nums, _limit, idx, _prev) when idx < 0, do: idx

  defp skip_backward(nums, limit, idx, prev) do
    if Enum.at(nums, idx) == prev do
      skip_backward(nums, limit, idx - 1, prev)
    else
      idx
    end
  end
end
```
