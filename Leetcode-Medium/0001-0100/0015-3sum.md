## Cpp

```cpp
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> res;
        int n = nums.size();
        if (n < 3) return res;
        sort(nums.begin(), nums.end());
        for (int i = 0; i < n - 2; ++i) {
            if (i > 0 && nums[i] == nums[i - 1]) continue;
            int left = i + 1, right = n - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum == 0) {
                    res.push_back({nums[i], nums[left], nums[right]});
                    ++left;
                    --right;
                    while (left < right && nums[left] == nums[left - 1]) ++left;
                    while (left < right && nums[right] == nums[right + 1]) --right;
                } else if (sum < 0) {
                    ++left;
                } else {
                    --right;
                }
            }
        }
        return res;
    }
};
```

```java
import java.util.*;

class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        if (nums == null || nums.length < 3) return res;
        Arrays.sort(nums);
        int n = nums.length;
        for (int i = 0; i < n - 2; i++) {
            if (i > 0 && nums[i] == nums[i - 1]) continue; // skip duplicate i
            int left = i + 1, right = n - 1;
            while (left < right) {
                long sum = (long) nums[i] + nums[left] + nums[right];
                if (sum == 0) {
                    res.add(Arrays.asList(nums[i], nums[left], nums[right]));
                    // skip duplicates for left and right
                    int curLeft = nums[left], curRight = nums[right];
                    while (left < right && nums[left] == curLeft) left++;
                    while (left < right && nums[right] == curRight) right--;
                } else if (sum < 0) {
                    left++;
                } else {
                    right--;
                }
            }
        }
        return res;
    }
}
```

```c
#include <stdlib.h>

static int cmp_int(const void *a, const void *b) {
    int va = *(const int *)a;
    int vb = *(const int *)b;
    if (va < vb) return -1;
    if (va > vb) return 1;
    return 0;
}

/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
int** threeSum(int* nums, int numsSize, int* returnSize, int*** returnColumnSizes) {
    if (numsSize < 3) {
        *returnSize = 0;
        *returnColumnSizes = NULL;
        return NULL;
    }

    qsort(nums, numsSize, sizeof(int), cmp_int);

    int capacity = 16;
    int **result = (int **)malloc(capacity * sizeof(int *));
    int *colSizes = (int *)malloc(capacity * sizeof(int));
    int count = 0;

    for (int i = 0; i < numsSize - 2; ++i) {
        if (i > 0 && nums[i] == nums[i - 1]) continue; // skip duplicate i

        int left = i + 1;
        int right = numsSize - 1;

        while (left < right) {
            long sum = (long)nums[i] + nums[left] + nums[right];
            if (sum == 0) {
                if (count == capacity) {
                    capacity <<= 1;
                    result = (int **)realloc(result, capacity * sizeof(int *));
                    colSizes = (int *)realloc(colSizes, capacity * sizeof(int));
                }
                int *triplet = (int *)malloc(3 * sizeof(int));
                triplet[0] = nums[i];
                triplet[1] = nums[left];
                triplet[2] = nums[right];
                result[count] = triplet;
                colSizes[count] = 3;
                ++count;

                int curLeft = nums[left];
                int curRight = nums[right];
                while (left < right && nums[left] == curLeft) ++left;
                while (left < right && nums[right] == curRight) --right;
            } else if (sum < 0) {
                ++left;
            } else {
                --right;
            }
        }
    }

    if (count == 0) {
        free(result);
        free(colSizes);
        *returnSize = 0;
        *returnColumnSizes = NULL;
        return NULL;
    }

    *returnSize = count;
    *returnColumnSizes = colSizes;
    return result;
}
```

```erlang
-spec three_sum(Nums :: [integer()]) -> [[integer()]].
three_sum(Nums) ->
    Sorted = lists:sort(Nums),
    Len = length(Sorted),
    Res = triplet_loop(0, Len, Sorted, []),
    lists:reverse(Res).

%% ------------------------------------------------------------------
%% Internal helpers
%% ------------------------------------------------------------------

-spec get_elem([integer()], non_neg_integer()) -> integer().
get_elem(List, Index) ->
    %% 0‑based index
    lists:nth(Index + 1, List).

-spec skip_forward(non_neg_integer(), [integer()], integer()) -> non_neg_integer().
skip_forward(Index, Sorted, Prev) ->
    Len = length(Sorted),
    if
        Index >= Len -> Index;
        true ->
            Val = get_elem(Sorted, Index),
            if
                Val == Prev -> skip_forward(Index + 1, Sorted, Prev);
                true -> Index
            end
    end.

-spec skip_backward(integer(), [integer()], integer()) -> integer().
skip_backward(Index, Sorted, Prev) when Index < 0 ->
    Index;
skip_backward(Index, Sorted, Prev) ->
    Val = get_elem(Sorted, Index),
    if
        Val == Prev -> skip_backward(Index - 1, Sorted, Prev);
        true -> Index
    end.

-spec two_pointer(non_neg_integer(), non_neg_integer(), integer(),
                  [integer()], integer(), [[integer()]]) -> [[integer()]].
two_pointer(_I, L, R, _Sorted, _Ai, Acc) when L >= R ->
    Acc;
two_pointer(I, L, R, Sorted, Ai, Acc) ->
    Aj = get_elem(Sorted, L),
    Ak = get_elem(Sorted, R),
    Sum = Ai + Aj + Ak,
    case Sum of
        0 ->
            Triplet = [Ai, Aj, Ak],
            NewL = skip_forward(L + 1, Sorted, Aj),
            NewR = skip_backward(R - 1, Sorted, Ak),
            two_pointer(I, NewL, NewR, Sorted, Ai, [Triplet | Acc]);
        S when S < 0 ->
            two_pointer(I, L + 1, R, Sorted, Ai, Acc);
        _ ->
            two_pointer(I, L, R - 1, Sorted, Ai, Acc)
    end.

-spec triplet_loop(non_neg_integer(), non_neg_integer(),
                  [integer()], [[integer()]]) -> [[integer()]].
triplet_loop(I, Len, _Sorted, Acc) when I >= Len - 2 ->
    Acc;
triplet_loop(I, Len, Sorted, Acc) ->
    Ai = get_elem(Sorted, I),
    Duplicate = (I > 0) andalso (Ai == get_elem(Sorted, I - 1)),
    case Duplicate of
        true ->
            triplet_loop(I + 1, Len, Sorted, Acc);
        false ->
            NewAcc = two_pointer(I, I + 1, Len - 1, Sorted, Ai, Acc),
            triplet_loop(I + 1, Len, Sorted, NewAcc)
```

## Elixir

```elixir
defmodule Solution do
  @spec three_sum(nums :: [integer]) :: [[integer]]
  def three_sum(nums) do
    sorted = Enum.sort(nums)
    arr = List.to_tuple(sorted)
    n = tuple_size(arr)

    0..(n - 3)
    |> Enum.reduce([], fn i, acc ->
      if i > 0 and elem(arr, i) == elem(arr, i - 1) do
        acc
      else
        ai = elem(arr, i)
        triples = two_sum(ai, arr, i + 1, n - 1, [])
        acc ++ triples
      end
    end)
  end

  defp two_sum(_ai, _arr, left, right, acc) when left >= right do
    Enum.reverse(acc)
  end

  defp two_sum(ai, arr, left, right, acc) do
    a_left = elem(arr, left)
    a_right = elem(arr, right)
    sum = ai + a_left + a_right

    cond do
      sum == 0 ->
        triplet = [ai, a_left, a_right]
        new_left = skip_forward(arr, left + 1, a_left)
        new_right = skip_backward(arr, right - 1, a_right)
        two_sum(ai, arr, new_left, new_right, [triplet | acc])

      sum < 0 ->
        new_left = skip_forward(arr, left + 1, a_left)
        two_sum(ai, arr, new_left, right, acc)

      true ->
        new_right = skip_backward(arr, right - 1, a_right)
        two_sum(ai, arr, left, new_right, acc)
    end
  end

  defp skip_forward(arr, idx, val) do
    n = tuple_size(arr)

    if idx < n and elem(arr, idx) == val do
      skip_forward(arr, idx + 1, val)
    else
      idx
    end
  end

  defp skip_backward(_arr, idx, _val) when idx < 0, do: idx

  defp skip_backward(arr, idx, val) do
    if elem(arr, idx) == val do
      skip_backward(arr, idx - 1, val)
    else
      idx
    end
  end
end
```
