```cpp
class Solution {
public:
    vector<vector<int>> updateMatrix(vector<vector<int>>& mat) {
        int m = mat.size();
        int n = mat[0].size();
        vector<vector<int>> dist(m, vector<int>(n, -1));
        queue<pair<int,int>> q;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (mat[i][j] == 0) {
                    dist[i][j] = 0;
                    q.emplace(i, j);
                }
            }
        }
        const int dirs[4][2] = {{-1,0},{1,0},{0,-1},{0,1}};
        while (!q.empty()) {
            auto [x, y] = q.front(); q.pop();
            for (auto &d : dirs) {
                int nx = x + d[0];
                int ny = y + d[1];
                if (nx < 0 || nx >= m || ny < 0 || ny >= n) continue;
                if (dist[nx][ny] != -1) continue;
                dist[nx][ny] = dist[x][y] + 1;
                q.emplace(nx, ny);
            }
        }
        return dist;
    }
};
```

```java
class Solution {
    public int[][] updateMatrix(int[][] mat) {
        int m = mat.length;
        int n = mat[0].length;
        int[][] dist = new int[m][n];
        boolean[][] visited = new boolean[m][n];
        java.util.ArrayDeque<int[]> q = new java.util.ArrayDeque<>();
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (mat[i][j] == 0) {
                    dist[i][j] = 0;
                    visited[i][j] = true;
                    q.offer(new int[]{i, j});
                } else {
                    dist[i][j] = Integer.MAX_VALUE;
                }
            }
        }
        
        int[][] dirs = {{1,0},{-1,0},{0,1},{0,-1}};
        while (!q.isEmpty()) {
            int[] cur = q.poll();
            int x = cur[0], y = cur[1];
            for (int[] d : dirs) {
                int nx = x + d[0];
                int ny = y + d[1];
                if (nx >= 0 && nx < m && ny >= 0 && ny < n && !visited[nx][ny]) {
                    dist[nx][ny] = dist[x][y] + 1;
                    visited[nx][ny] = true;
                    q.offer(new int[]{nx, ny});
                }
            }
        }
        return dist;
    }
}
```

```python
        rows = len(mat)
        cols = len(mat[0])

            for j in range(cols):
```

```c
#include <stdlib.h>
#include <limits.h>

int** updateMatrix(int** mat, int matSize, int* matColSize, int* returnSize, int** returnColumnSizes) {
    int rows = matSize;
    if (rows == 0) {
        *returnSize = 0;
        *returnColumnSizes = NULL;
        return NULL;
    }
    int cols = matColSize[0]; // assume all rows have same length

    // Allocate result matrix
    int **dist = (int **)malloc(rows * sizeof(int *));
    for (int i = 0; i < rows; ++i) {
        dist[i] = (int *)malloc(cols * sizeof(int));
    }

    // Allocate return column sizes array
    int *colSizes = (int *)malloc(rows * sizeof(int));
    for (int i = 0; i < rows; ++i) colSizes[i] = cols;
    *returnColumnSizes = colSizes;
    *returnSize = rows;

    // Queue for BFS
    int capacity = rows * cols;
    int *qr = (int *)malloc(capacity * sizeof(int));
    int *qc = (int *)malloc(capacity * sizeof(int));
    int front = 0, back = 0;

    // Initialize distances and queue with all zero cells
    for (int i = 0; i < rows; ++i) {
        for (int j = 0; j < cols; ++j) {
            if (mat[i][j] == 0) {
                dist[i][j] = 0;
                qr[back] = i;
                qc[back] = j;
                back++;
            } else {
                dist[i][j] = INT_MAX;
            }
        }
    }

    // Directions: up, down, left, right
    const int dr[4] = {-1, 1, 0, 0};
    const int dc[4] = {0, 0, -1, 1};

    // BFS propagation
    while (front < back) {
        int r = qr[front];
        int c = qc[front];
        front++;
        int curDist = dist[r][c];

        for (int d = 0; d < 4; ++d) {
            int nr = r + dr[d];
            int nc = c + dc[d];
            if (nr >= 0 && nr < rows && nc >= 0 && nc < cols) {
                if (dist[nr][nc] > curDist + 1) {
                    dist[nr][nc] = curDist + 1;
                    qr[back] = nr;
                    qc[back] = nc;
                    back++;
                }
            }
        }
    }

    free(qr);
    free(qc);
    return dist;
}
```

```go
func updateMatrix(mat [][]int) [][]int {
	rows := len(mat)
	cols := len(mat[0])
	dist := make([][]int, rows)
	for i := 0; i < rows; i++ {
		dist[i] = make([]int, cols)
		for j := 0; j < cols; j++ {
			if mat[i][j] == 0 {
				dist[i][j] = 0
			} else {
				dist[i][j] = -1
			}
		}
	}

	type cell struct{ r, c int }
	queue := make([]cell, 0)
	for i := 0; i < rows; i++ {
		for j := 0; j < cols; j++ {
			if mat[i][j] == 0 {
				queue = append(queue, cell{i, j})
			}
		}
	}

	dirs := []cell{{-1, 0}, {1, 0}, {0, -1}, {0, 1}}

	for head < len(queue) {

			nr, nc := cur.r+d.r, cur.c+d.c
			if nr >= 0 && nr < rows && nc >= 0 && nc < cols && dist[nr][nc] == -1 {
				dist[nr][nc] = dist[cur.r][cur.c] + 1
				queue = append(queue, cell{nr, nc})
			}
		}
	}

	return dist
```

```rust
use std::collections::VecDeque;

pub struct Solution;

impl Solution {
    pub fn update_matrix(mat: Vec<Vec<i32>>) -> Vec<Vec<i32>> {
        let m = mat.len();
        if m == 0 {
            return vec![];
        }
        let n = mat[0].len();

        let mut dist = vec![vec![-1i32; n]; m];
        let mut q: VecDeque<(usize, usize)> = VecDeque::new();

        for i in 0..m {
            for j in 0..n {
                if mat[i][j] == 0 {
                    dist[i][j] = 0;
                    q.push_back((i, j));
                }
            }
        }

        let dirs = [(1_i32, 0_i32), (-1, 0), (0, 1), (0, -1)];
        while let Some((i, j)) = q.pop_front() {
            let cur = dist[i][j];
            for &(dx, dy) in &dirs {
                let ni = i as i32 + dx;
                let nj = j as i32 + dy;
                if ni >= 0 && ni < m as i32 && nj >= 0 && nj < n as i32 {
                    let ui = ni as usize;
                    let uj = nj as usize;
                    if dist[ui][uj] == -1 {
                        dist[ui][uj] = cur + 1;
                        q.push_back((ui, uj));
                    }
                }
            }
        }

        dist
    }
```

```racket
  (-> (listof (listof exact-integer?)) (listof (listof exact-integer?)))

    (define q (make-queue))

      (for ([j (in-range cols)])
        (when (= (vector-ref (vector-ref grid i) j) 0)
          (vector-set! (vector-ref dist i) j 0)
          (enqueue! q (list i j)))))
    ;; BFS propagation
    (let bfs ()
      (unless (queue-empty? q)
        (define pos (dequeue! q))
        (define i (first pos))
        (define j (second pos))
        (define d (vector-ref (vector-ref dist i) j))

          (reverse acc)
```

```elixir
      Enum.reduce(Enum.with_index(mat), {:queue.new(), %{}}, fn {row, r},
                                                                {q_acc, dm_acc} ->
        {new_q, new_dm} =
          Enum.reduce(Enum.with_index(row), {q_acc, dm_acc}, fn {val, c},
                                                               {qq, dmm} ->
            if val == 0 do
              {:queue.in({r, c}, qq), Map.put(dmm, {r, c}, 0)}
            else
              {qq, dmm}
            end
          end)

        {new_q, new_dm}
      end)

    bfs(queue, dist_map, rows, cols)
  end

  defp bfs(queue, dist_map, rows, cols) do
    case :queue.out(queue) do
      {:empty, _} ->
        for r <- 0..(rows - 1) do
          for c <- 0..(cols - 1), into: [] do
            Map.get(dist_map, {r, c})
          end
        end

      {{:value, {r, c}}, q_rest} ->
        cur_dist = Map.get(dist_map, {r, c})

        neighbors = [{r - 1, c}, {r + 1, c}, {r, c - 1}, {r, c + 1}]

        {new_q, new_dm} =
          Enum.reduce(neighbors, {q_rest, dist_map}, fn {nr, nc},
                                                       {qq, dmm} ->
            if nr >= 0 and nr < rows and nc >= 0 and nc < cols do
              if Map.has_key?(dmm, {nr, nc}) do
                {qq, dmm}
              else
                nd = cur_dist + 1
                {:queue.in({nr, nc}, qq), Map.put(dmm, {nr, nc}, nd)}
              end
            else
              {qq, dmm}
            end
          end)

        bfs(new_q, new_dm, rows, cols)
    end
  end
end
```
