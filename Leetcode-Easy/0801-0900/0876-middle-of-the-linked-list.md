# 0876. Middle of the Linked List

## Cpp

```cpp
/ **
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        ListNode* slow = head;
        ListNode* fast = head;
        while (fast && fast->next) {
            slow = slow->next;
            fast = fast->next->next;
        }
        return slow;
    }
};
```

impl Solution {
    pub fn middle_node(head: Option<Box<ListNode>>) -> Option<Box<ListNode>> {
        let mut slow = &head;
        let mut fast = &head;
        while let Some(f_node) = fast {
            if f_node.next.is_none() {
                break;
            }
            fast = &f_node.next.as_ref().unwrap().next;
            slow = &slow.as_ref().unwrap().next;

%% Definition for singly-linked list.
%%
%% -record(list_node, {val = 0 :: integer(),
%%                     next = null :: 'null' | #list_node{}}).

-spec middle_node(Head :: #list_node{} | null) -> #list_node{} | null.
middle_node(null) ->
    null;
middle_node(Head) ->
    middle_node_loop(Head, Head).

middle_node_loop(Slow, Fast) ->
    case Fast of
        #list_node{next = Next} ->
            case Next of
                #list_node{} = NextNode ->
                    NewSlow = Slow#list_node.next,
                    NewFast = NextNode#list_node.next,
                    middle_node_loop(NewSlow, NewFast);
                null ->
                    Slow
            end;
        null ->
            Slow
    end.
