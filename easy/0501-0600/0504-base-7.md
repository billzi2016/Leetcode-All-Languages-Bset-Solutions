# 0504. Base 7

## Cpp

```cpp
class Solution {
public:
    string convertToBase7(int num) {
        if (num == 0) return "0";
        bool negative = num < 0;
        long long n = num;
        if (negative) n = -n;
        string res;
        while (n > 0) {
            int digit = n % 7;
            res.push_back('0' + digit);
            n /= 7;
        }
        if (negative) res.push_back('-');
        reverse(res.begin(), res.end());
        return res;
    }
};
```

char* convertToBase7(int num) {
    if (num == 0) {
        char *zero = (char *)malloc(2);
        zero[0] = '0';
        zero[1] = '\0';
        return zero;
    }
    
    int negative = num < 0;
    long long n = num;
    if (negative) n = -n;  // make positive
    
    char buf[12]; // enough for sign, digits and null
    int idx = 0;
    
    while (n > 0) {
        int digit = n % 7;
        buf[idx++] = '0' + digit;
        n /= 7;
    }
    
    if (negative) {
        buf[idx++] = '-';
    }
    
    char *res = (char *)malloc(idx + 1);
    for (int i = 0; i < idx; ++i) {
        res[i] = buf[idx - 1 - i];
    }
    res[idx] = '\0';
    return res;

impl Solution {
    pub fn convert_to_base7(num: i32) -> String {
        if num == 0 {
            return "0".to_string();
        }
        let negative = num < 0;
        let mut n = if negative { -(num as i64) } else { num as i64 };
        let mut chars = Vec::new();
        while n > 0 {
            let digit = (n % 7) as u8;
            chars.push((b'0' + digit) as char);
            n /= 7;
        }
        chars.reverse();
        let mut result: String = chars.iter().collect();
        if negative {
            result.insert(0, '-');
        }
        result
    }
}

            Sign = if Num < 0 -> "-"; true -> "" end,
            Abs = abs(Num),
            Digits = digits_rev(Abs, []),
            BinDigits = list_to_binary(lists:reverse(Digits)),

    R = N rem 7,
    Char = $0 + R,

    sign = if num < 0, do: "-", else: ""
