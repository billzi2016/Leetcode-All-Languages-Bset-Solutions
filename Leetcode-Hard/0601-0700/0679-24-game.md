class Solution {
public:
    bool judgePoint24(vector<int>& cards) {
        vector<double> nums;
        for (int c : cards) nums.push_back(c);
        return dfs(nums);
    }
private:
    const double TARGET = 24.0;
    const double EPS = 1e-6;
    
    bool dfs(vector<double>& nums) {
        int n = nums.size();
        if (n == 1) {
            return fabs(nums[0] - TARGET) < EPS;
        }
        // try all pairs
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                double a = nums[i], b = nums[j];
                vector<double> next;
                for (int k = 0; k < n; ++k) {
                    if (k != i && k != j) next.push_back(nums[k]);
                }
                
                // addition
                next.push_back(a + b);
                if (dfs(next)) return true;
                next.pop_back();
                
                // subtraction a - b
                next.push_back(a - b);
                if (dfs(next)) return true;
                next.pop_back();
                
                // subtraction b - a
                next.push_back(b - a);
                if (dfs(next)) return true;
                next.pop_back();
                
                // multiplication
                next.push_back(a * b);
                if (dfs(next)) return true;
                next.pop_back();
                
                // division a / b
                if (fabs(b) > EPS) {
                    next.push_back(a / b);
                    if (dfs(next)) return true;
                    next.pop_back();
                }
                // division b / a
                if (fabs(a) > EPS) {
                    next.push_back(b / a);
                    if (dfs(next)) return true;
                    next.pop_back();
                }
            }
        }
        return false;
    }

    private static final double EPS = 1e-6;

    public boolean judgePoint24(int[] cards) {
        List<Double> nums = new ArrayList<>();
        for (int c : cards) {
            nums.add((double) c);

                double b = nums.get(j);

                    if (k != i && k != j) {
                        next.add(nums.get(k));

                next.add(a + b);
                if (dfs(next)) return true;
                next.remove(next.size() - 1);

                // subtraction a - b
                next.add(a - b);

                    next.add(a / b);

                }
            }
        }

class Solution(object):
    def judgePoint24(self, cards):
        """
        :type cards: List[int]
        :rtype: bool
        """
        target = 24.0
        eps = 1e-6

        def helper(nums):
            if len(nums) == 1:
                return abs(nums[0] - target) < eps
            n = len(nums)
            for i in range(n):
                for j in range(n):
                    if i == j:
                        continue
                    a, b = nums[i], nums[j]
                    next_nums = [nums[k] for k in range(n) if k != i and k != j]

                    # addition (commutative, do once)
                    if i < j:  # avoid duplicate for + and *
                        next_nums.append(a + b)
                        if helper(next_nums):
                            return True
                        next_nums.pop()

                    # multiplication (commutative, do once)
                    if i < j:
                        next_nums.append(a * b)
                        if helper(next_nums):
                            return True
                        next_nums.pop()

                    # subtraction a - b
                    next_nums.append(a - b)
                    if helper(next_nums):
                        return True
                    next_nums.pop()

                    # division a / b
                    if abs(b) > eps:
                        next_nums.append(a / b)
                        if helper(next_nums):
                            return True
                        next_nums.pop()
            return False

        nums = [float(x) for x in cards]
        return helper(nums)

        for (int j = 0; j < n; ++j) {

                if (k != i && k != j) {
                    next[idx++] = nums[k];
                }
            }
            // addition
            next[idx] = a + b;
            if (solve(next, idx + 1)) return true;

            next[idx] = a - b;

            if (fabs(b) > EPS) {
                next[idx] = a / b;
                if (solve(next, idx + 1)) return true;
            }
        }
    }
    return false;
}

package main

import "math"

func judgePoint24(cards []int) bool {
	nums := make([]float64, 4)
	for i, v := range cards {
		nums[i] = float64(v)
	}
	return dfs(nums)
}

func dfs(vals []float64) bool {
	if len(vals) == 1 {
		return math.Abs(vals[0]-24.0) < 1e-6
	}
	n := len(vals)
	for i := 0; i < n; i++ {
		for j := 0; j < n; j++ {
			if i == j {
				continue
			}
			a, b := vals[i], vals[j]

			// collect remaining numbers
			rest := make([]float64, 0, n-1)
			for k := 0; k < n; k++ {
				if k != i && k != j {
					rest = append(rest, vals[k])
				}
			}

			candidates := []float64{a + b, a - b, b - a, a * b}
			if math.Abs(b) > 1e-9 {
				candidates = append(candidates, a/b)
			}
			if math.Abs(a) > 1e-9 {
				candidates = append(candidates, b/a)
			}

			for _, c := range candidates {
				next := make([]float64, 0, n-1)
				next = append(next, c)
				next = append(next, rest...)
				if dfs(next) {
					return true
				}
			}
		}
	}
	return false
}

      end
    end

impl Solution {
    pub fn judge_point24(cards: Vec<i32>) -> bool {
        let nums: Vec<f64> = cards.into_iter().map(|x| x as f64).collect();
        Self::dfs(nums)
    }

    fn dfs(nums: Vec<f64>) -> bool {
        const EPS: f64 = 1e-6;
        if nums.len() == 1 {
            return (nums[0] - 24.0).abs() < EPS;
        }
        let n = nums.len();
        for i in 0..n {
            for j in i + 1..n {
                // remaining numbers after removing i and j
                let mut rest: Vec<f64> = Vec::with_capacity(n - 1);
                for k in 0..n {
                    if k != i && k != j {
                        rest.push(nums[k]);
                    }
                }

                let b = nums[j];

                // all possible results of applying an operator to a and b
                let mut candidates: Vec<f64> = Vec::new();
                candidates.push(a + b);
                candidates.push(a - b);
                candidates.push(b - a);
                candidates.push(a * b);
                if b.abs() > EPS {
                    candidates.push(a / b);
                }
                if a.abs() > EPS {
                    candidates.push(b / a);
                }

                for val in candidates {
                    let mut next = rest.clone();
                    next.push(val);
                    if Self::dfs(next) {
                        return true;
                    }
                }
            }
        }
        false
    }

    abs(x - 24.0) < 1.0e-6
  end

  defp solve(list) do
    len = length(list)

    Enum.any?(0..(len - 2), fn i ->
      Enum.any?((i + 1)..(len - 1), fn j ->
        a = Enum.at(list, i)
        b = Enum.at(list, j)

        rest =
          list
          |> List.delete_at(j)
          |> List.delete_at(i)

        possible_results(a, b)
        |> Enum.any?(fn r -> solve([r | rest]) end)
      end)
    end)
  end

  defp possible_results(a, b) do
    results = [a + b, a * b, a - b, b - a]

    results =
      if abs(b) > 1.0e-6 do
        [a / b | results]
      else
        results
      end

    results =
      if abs(a) > 1.0e-6 do
        [b / a | results]
      else
        results
      end

    results
  end
end
