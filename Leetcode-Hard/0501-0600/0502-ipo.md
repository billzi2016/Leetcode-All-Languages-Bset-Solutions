```cpp
class Solution {
public:
    int findMaximizedCapital(int k, int w, vector<int>& profits, vector<int>& capital) {
        int n = profits.size();
        vector<pair<int,int>> projects;
        projects.reserve(n);
        for (int i = 0; i < n; ++i) {
            projects.emplace_back(capital[i], profits[i]);
        }
        sort(projects.begin(), projects.end(),
             [](const pair<int,int>& a, const pair<int,int>& b){ return a.first < b.first; });
        
        priority_queue<int> maxProfits;
        int idx = 0;
        for (int i = 0; i < k; ++i) {
            while (idx < n && projects[idx].first <= w) {
                maxProfits.push(projects[idx].second);
                ++idx;
            }
            if (maxProfits.empty()) break;
            w += maxProfits.top();
            maxProfits.pop();
        }
        return w;
    }
};
```

```python
class Solution(object):
    def findMaximizedCapital(self, k, w, profits, capital):
        """
        :type k: int
        :type w: int
        :type profits: List[int]
        :type capital: List[int]
        :rtype: int
        """
        import heapq

        # Pair each project's required capital with its profit
        projects = list(zip(capital, profits))
        # Sort by required capital ascending
        projects.sort(key=lambda x: x[0])

        max_profit_heap = []
        idx = 0
        n = len(projects)

        for _ in range(k):
            # Add all affordable projects to the max-heap
            while idx < n and projects[idx][0] <= w:
                heapq.heappush(max_profit_heap, -projects[idx][1])
                idx += 1

            if not max_profit_heap:
                break  # No more projects can be undertaken

            # Choose the most profitable affordable project
            w += -heapq.heappop(max_profit_heap)

        return w
```

```c
#include <stdlib.h>

typedef struct {
    int capital;
    int profit;
} Project;

static int compareProject(const void *a, const void *b) {
    const Project *pa = (const Project *)a;
    const Project *pb = (const Project *)b;
    return pa->capital - pb->capital;
}

/* max-heap operations for integers */
static void heapPush(int *heap, int *size, int value) {
    int idx = (*size)++;
    heap[idx] = value;
    while (idx > 0) {
        int parent = (idx - 1) >> 1;
        if (heap[parent] >= heap[idx]) break;
        int tmp = heap[parent];
        heap[parent] = heap[idx];
        heap[idx] = tmp;
        idx = parent;
    }
}

static int heapPop(int *heap, int *size) {
    int top = heap[0];
    heap[0] = heap[--(*size)];
    int idx = 0;
    while (1) {
        int left = (idx << 1) + 1;
        if (left >= *size) break;
        int right = left + 1;
        int largest = left;
        if (right < *size && heap[right] > heap[left]) largest = right;
        if (heap[largest] <= heap[idx]) break;
        int tmp = heap[largest];
        heap[largest] = heap[idx];
        heap[idx] = tmp;
        idx = largest;
    }
    return top;
}

int findMaximizedCapital(int k, int w, int* profits, int profitsSize, int* capital, int capitalSize) {
    int n = profitsSize;
    Project *projects = (Project *)malloc(n * sizeof(Project));
    for (int i = 0; i < n; ++i) {
        projects[i].capital = capital[i];
        projects[i].profit = profits[i];
    }
    qsort(projects, n, sizeof(Project), compareProject);

    int *heap = (int *)malloc(n * sizeof(int));
    int heapSize = 0;
    int idx = 0;

    for (int i = 0; i < k; ++i) {
        while (idx < n && projects[idx].capital <= w) {
            heapPush(heap, &heapSize, projects[idx].profit);
            ++idx;
        }
        if (heapSize == 0) break;
        w += heapPop(heap, &heapSize);
    }

    free(projects);
    free(heap);
    return w;
}
```

```javascript
class MaxHeap {
    constructor() {
        this.heap = [];
    }
    size() {
        return this.heap.length;
    }
    push(val) {
        const h = this.heap;
        h.push(val);
        let i = h.length - 1;
        while (i > 0) {
            const p = (i - 1) >> 1;
            if (h[p] >= h[i]) break;
            [h[p], h[i]] = [h[i], h[p]];
            i = p;
        }
    }
    pop() {
        const h = this.heap;
        if (h.length === 0) return undefined;
        const max = h[0];
        const end = h.pop();
        if (h.length > 0) {
            h[0] = end;
            let i = 0;
            while (true) {
                let l = i * 2 + 1,
                    r = l + 1,
                    largest = i;
                if (l < h.length && h[l] > h[largest]) largest = l;
                if (r < h.length && h[r] > h[largest]) largest = r;
                if (largest === i) break;
                [h[i], h[largest]] = [h[largest], h[i]];
                i = largest;
            }
        }
        return max;
    }
}
```

```swift
struct MaxHeap {
    private var heap: [Int] = []
    
    var isEmpty: Bool { heap.isEmpty }
    
    mutating func push(_ value: Int) {
        heap.append(value)
        siftUp(heap.count - 1)
    }
    
    mutating func pop() -> Int? {
        guard !heap.isEmpty else { return nil }
        if heap.count == 1 {
            return heap.removeLast()
        }
        let top = heap[0]
        heap[0] = heap.removeLast()
        siftDown(0)
        return top
    }
    
    private mutating func siftUp(_ index: Int) {
        var child = index
        while child > 0 {
            let parent = (child - 1) / 2
            if heap[parent] < heap[child] {
                heap.swapAt(parent, child)
                child = parent
            } else {
                break
            }
        }
    }
    
    private mutating func siftDown(_ index: Int) {
        var parent = index
        while true {
            let left = parent * 2 + 1
            let right = left + 1
            var largest = parent
            if left < heap.count && heap[left] > heap[largest] {
                largest = left
            }
            if right < heap.count && heap[right] > heap[largest] {
                largest = right
            }
            if largest == parent { break }
            heap.swapAt(parent, largest)
            parent = largest
        }
    }
}
```

```dart
  int findMaximizedCapital(int k, int w, List<int> profits, List<int> capital) {
    int n = profits.length;
    List<int> order = List.generate(n, (i) => i);
    order.sort((a, b) => capital[a].compareTo(capital[b]));
    List<int> maxHeap = [];

    for (int i = 0; i < k; i++) {
      while (idx < n && capital[order[idx]] <= cur) {
        _push(maxHeap, profits[order[idx]]);
        idx++;
      }
      if (maxHeap.isEmpty) break;
      cur += _pop(maxHeap);
    }
    return cur;
  }

  void _push(List<int> heap, int val) {
    heap.add(val);
    int i = heap.length - 1;
    while (i > 0) {
      int p = (i - 1) >> 1;
      if (heap[p] >= heap[i]) break;
      int tmp = heap[p];
      heap[p] = heap[i];
      heap[i] = tmp;
      i = p;
    }
  }

  int _pop(List<int> heap) {
    int res = heap[0];
    int last = heap.removeLast();
    if (heap.isNotEmpty) {
      heap[0] = last;
      int n = heap.length;
      int i = 0;
      while (true) {
        int left = i * 2 + 1;
        int right = left + 1;

        if (right < n && heap[right] > heap[largest]) largest = right;
        if (largest == i) break;
        int tmp = heap[i];
        heap[i] = heap[largest];
        heap[largest] = tmp;
        i = largest;
      }
    }
    return res;
  }
```

```scala
object Solution {
    def findMaximizedCapital(k: Int, w: Int, profits: Array[Int], capital: Array[Int]): Int = {
        val n = profits.length
        // indices sorted by required capital ascending
        val order = (0 until n).toArray.sortBy(i => capital(i))
        var idx = 0
        var curW: Long = w.toLong
        import scala.collection.mutable.PriorityQueue
        implicit val maxOrdering: Ordering[Int] = Ordering.Int.reverse // for max-heap
        val pq = PriorityQueue.empty[Int]

        var remaining = k
        while (remaining > 0) {
            while (idx < n && capital(order(idx)) <= curW) {
                pq.enqueue(profits(order(idx)))
                idx += 1
            }
            if (pq.isEmpty) {
                // No more affordable projects
                return curW.toInt
            }
            curW += pq.dequeue()
            remaining -= 1
        }
        curW.toInt
    }
```

```erlang
    loop(K, W, Sorted, Tree0).

loop(0, W, _Projects, _Tree) ->
    W;
loop(K, W, Projects, Tree) ->
    {RemProjects, Tree1} = add_affordable(Projects, W, Tree),
    case gb_trees:is_empty(Tree1) of
        true -> W;
        false ->
            {Profit, NewTree} = heap_pop_max(Tree1),
            loop(K - 1, W + Profit, RemProjects, NewTree)
    end.

add_affordable([], _W, Tree) ->
    {[], Tree};
add_affordable([{Cap, Prof}=H|Tail], W, Tree) when Cap =< W ->
    Tree2 = heap_insert(Tree, Prof),
    add_affordable(Tail, W, Tree2);
add_affordable(Rest, _W, Tree) ->
    {Rest, Tree}.

heap_insert(Tree, Profit) ->
    case gb_trees:is_defined(Profit, Tree) of
        true ->
            Count = gb_trees:get(Profit, Tree),
            gb_trees:update(Profit, Count + 1, Tree);
        false ->
            gb_trees:insert(Profit, 1, Tree)
    end.

heap_pop_max(Tree) ->
    {Profit, Count} = gb_trees:largest(Tree),
    NewTree =
        if
            Count > 1 ->
                gb_trees:update(Profit, Count - 1, Tree);
            true ->
                gb_trees:delete(Profit, Tree)
        end,
    {Profit, NewTree}.
```

```elixir
defmodule Solution do
  @spec find_maximized_capital(k :: integer, w :: integer, profits :: [integer], capital :: [integer]) :: integer
  def find_maximized_capital(k, w, profits, capital) do
    projects = Enum.zip([capital, profits])
    sorted = Enum.sort_by(projects, fn {c, _p} -> c end)
    iterate(k, w, 0, sorted, :gb_trees.empty())
  end

  defp iterate(0, w, _idx, _sorted, _heap), do: w

  defp iterate(k_rem, w, idx, sorted, heap) do
    {new_idx, new_heap} = add_affordable(sorted, idx, w, heap)

    if :gb_trees.is_empty(new_heap) do
      w
    else
      {profit, _cnt} = :gb_trees.largest(new_heap)

      updated_heap =
        case :gb_trees.lookup(profit, new_heap) do
          {:value, 1} -> :gb_trees.delete(profit, new_heap)
          {:value, cnt} -> :gb_trees.update(profit, cnt - 1, new_heap)
        end

      iterate(k_rem - 1, w + profit, new_idx, sorted, updated_heap)
    end
  end

  defp add_affordable(sorted, idx, w, heap) do
    n = length(sorted)

    if idx < n and elem(Enum.at(sorted, idx), 0) <= w do
      {c, p} = Enum.at(sorted, idx)

      new_heap =
        case :gb_trees.lookup(p, heap) do
          :none -> :gb_trees.insert(p, 1, heap)
          {:value, cnt} -> :gb_trees.update(p, cnt + 1, heap)
        end

      add_affordable(sorted, idx + 1, w, new_heap)
    else
      {idx, heap}
    end
  end
end
```
